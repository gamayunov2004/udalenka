cat > calc_syscall_percent.sh << 'EOF'
#!/usr/bin/env bash

if [[ $# -ne 2 ]]; then
    echo "Usage: ./calc_syscall_percent.sh TRACE_FILE PROGRAM_TIME_SEC"
    exit 1
fi

trace_file="$1"
program_time="$2"

if [[ ! -f "$trace_file" ]]; then
    echo "Ошибка: файл трассировки '$trace_file' не найден"
    exit 2
fi

awk -v program_time="$program_time" '
{
    while (match($0, /<[0-9]+\.[0-9]+>/)) {
        syscall_time += substr($0, RSTART + 1, RLENGTH - 2)
        $0 = substr($0, RSTART + RLENGTH)
    }
}
END {
    if (program_time <= 0) {
        print "Ошибка: общее время программы должно быть больше нуля"
        exit 3
    }
    percent = syscall_time / program_time * 100
    printf "Суммарное время системных вызовов: %.6f сек\n", syscall_time
    printf "Общее время программы: %.2f сек\n", program_time
    printf "Доля времени системных вызовов: %.4f%%\n", percent
}' "$trace_file"
EOF

chmod +x calc_syscall_percent.sh
./calc_syscall_percent.sh own_trace.log 62
