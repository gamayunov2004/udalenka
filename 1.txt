sudo tee /etc/systemd/system/lab12-memmap.service > /dev/null <<'EOF'
[Unit]
Description=Lab12 memory map collector

[Service]
Type=oneshot
EnvironmentFile=/etc/lab12_mem.env
ExecStart=/usr/local/bin/collect_mem_map.sh $TARGET_PID $TARGET_MODE
EOF



sudo tee /etc/systemd/system/lab12-memmap.timer > /dev/null <<'EOF'
[Unit]
Description=Run Lab12 memory map collector every 30 seconds

[Timer]
OnActiveSec=5s
OnUnitActiveSec=30s
Unit=lab12-memmap.service

[Install]
WantedBy=timers.target
EOF
