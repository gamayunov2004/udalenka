--подготовка--
mkdir -p ~/Gamayunov/pr3_task1
cd ~/Gamayunov/pr3_task1
sudo apt update
sudo apt install -y apache2 nginx curl
apache2 -v
nginx -v

--1.1--
sudo cp /etc/apache2/ports.conf /etc/apache2/ports.conf.bak

sudo tee /etc/apache2/ports.conf > /dev/null <<'EOF'
Listen 8081
Listen 8082
Listen 8083
EOF

cat /etc/apache2/ports.conf

--1.2--
sudo mkdir -p /var/www/server1 /var/www/server2 /var/www/server3

echo '<h1>Server 1</h1><p>Apache backend on port 8081</p>' | sudo tee /var/www/server1/index.html

echo '<h1>Server 2</h1><p>Apache backend on port 8082</p>' | sudo tee /var/www/server2/index.html

echo '<h1>Server 3</h1><p>Apache backend on port 8083</p>' | sudo tee /var/www/server3/index.html

ls -l /var/www/server1 /var/www/server2 /var/www/server3

--1.3--
sudo tee /etc/apache2/sites-available/server1.conf > /dev/null <<'EOF'
<VirtualHost *:8081>
    ServerName server1.local
    DocumentRoot /var/www/server1
    ErrorLog ${APACHE_LOG_DIR}/server1_error.log
    CustomLog ${APACHE_LOG_DIR}/server1_access.log combined
</VirtualHost>
EOF

sudo tee /etc/apache2/sites-available/server2.conf > /dev/null <<'EOF'
<VirtualHost *:8082>
    ServerName server2.local
    DocumentRoot /var/www/server2
    ErrorLog ${APACHE_LOG_DIR}/server2_error.log
    CustomLog ${APACHE_LOG_DIR}/server2_access.log combined
</VirtualHost>
EOF

sudo tee /etc/apache2/sites-available/server3.conf > /dev/null <<'EOF'
<VirtualHost *:8083>
    ServerName server3.local
    DocumentRoot /var/www/server3
    ErrorLog ${APACHE_LOG_DIR}/server3_error.log
    CustomLog ${APACHE_LOG_DIR}/server3_access.log combined
</VirtualHost>
EOF

sudo a2dissite 000-default.conf
sudo a2ensite server1.conf server2.conf server3.conf
sudo apache2ctl configtest
sudo systemctl restart apache2
sudo systemctl status apache2 --no-pager

--1.4--
curl -s http://localhost:8081
curl -s http://localhost:8082
curl -s http://localhost:8083
sudo ss -tlnp | grep apache2

--1.5--
sudo tee /etc/nginx/sites-available/pr3_balancer > /dev/null <<'EOF'
upstream apache_backend {
    server 127.0.0.1:8081;
    server 127.0.0.1:8082;
    server 127.0.0.1:8083;
}

server {
    listen 80 default_server;
    server_name _;

    location / {
        proxy_pass http://apache_backend;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }
}
EOF

sudo ln -sf /etc/nginx/sites-available/pr3_balancer /etc/nginx/sites-enabled/pr3_balancer
sudo rm -f /etc/nginx/sites-enabled/default
sudo nginx -t
sudo systemctl restart nginx
sudo systemctl status nginx --no-pager
sudo ss -tlnp | grep nginx

--1.6--
curl -s http://localhost
curl -s http://localhost
curl -s http://localhost
for i in {1..9}; do curl -s http://localhost | grep -o 'Server [123]'; done
